CREATE TABLE PUBLIC.CARGO (
	ID BIGINT NOT NULL,
	CARGO_TYPE VARCHAR(100) NOT NULL,
	NAME VARCHAR(100),
	WEIGHT INTEGER,
	CLOTHES_SIZE VARCHAR(100),
	CLOTHES_MATERIAL VARCHAR(100),
	STORE_TEMPERATURE INTEGER,
	EXPIRY_DATE TIMESTAMP,
	CONSTRAINT CARGO_PK PRIMARY KEY (ID)
);

CREATE TABLE PUBLIC.CARRIER (
	ID BIGINT NOT NULL,
	CARRIER_TYPE VARCHAR(100) NOT NULL,
	NAME VARCHAR(100),
	ADDRESS VARCHAR(100),
	CONSTRAINT CARRIER_PK PRIMARY KEY (ID)
);

CREATE TABLE PUBLIC.TRANSPORTATIONS (
	ID BIGINT NOT NULL,
	CARGO_REF BIGINT NOT NULL,
	CARRIER_REF BIGINT NOT NULL,
	DESCRIPTION VARCHAR(100),
	BILL_TO VARCHAR(100),
	BEGIN_DATE TIMESTAMP,
	CONSTRAINT TRANSPORTATIONS_PK PRIMARY KEY (ID),
	CONSTRAINT TRANSPORTATIONS_CARGO_FK_ FOREIGN KEY (CARGO_REF) REFERENCES PUBLIC.CARGO(ID),
	CONSTRAINT TRANSPORTATIONS_CARIER_FK FOREIGN KEY (CARRIER_REF) REFERENCES PUBLIC.CARRIER(ID)
);
